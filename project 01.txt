import pandas as pd
import matplotlib.pyplot as plt

# Read dataset
try:
    data = pd.read_csv("weather.csv")
except FileNotFoundError:
    print("Error: weather.csv file nahi mili. Same folder me file rakho.")
    exit()

# Convert Date column to datetime
data["Date"] = pd.to_datetime(data["Date"])

# Create Year and Month columns
data["Year"] = data["Date"].dt.year
data["Month"] = data["Date"].dt.month

# Categorize temperature
def categorize(temp):
    if temp <= 0:
        return "Freezing"
    elif temp < 30:
        return "Normal"
    else:
        return "Boiling"

data["Category"] = data["Temperature"].apply(categorize)

# Find hottest and coldest months
monthly_avg = data.groupby("Month")["Temperature"].mean()

hottest_month = monthly_avg.idxmax()
coldest_month = monthly_avg.idxmin()

print("Hottest Month:", hottest_month)
print("Coldest Month:", coldest_month)

# Yearly temperature trend
yearly_avg = data.groupby("Year")["Temperature"].mean()

# Plot graph
plt.plot(yearly_avg.index, yearly_avg.values)
plt.title("10-Year Temperature Trend")
plt.xlabel("Year")
plt.ylabel("Average Temperature (°C)")
plt.show()
